Kleeblattsuche im Wald I

Solange Kara nicht auf einem Kleeblatt steht, versucht er, gerade aus zu laufen. Falls vor ihm ein Baum ist, umläuft er ihn.

public void myProgram() {
  while (! kara.onLeaf()) {
    if (kara.treeFront()) {
      kara.turnLeft();
      kara.move();
      kara.turnRight();
      kara.move();
      kara.move();
        kara.turnRight();
        kara.move();
        kara.turnLeft();
    }
    else {
      kara.move();
    }
  }
  kara.removeLeaf();
}
